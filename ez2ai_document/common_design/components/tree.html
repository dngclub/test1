function TreeNode({ node, active, onSelect }) {
  const [open, setOpen] = useState(true)
  return (
    <>
      <div className={`xx-tree-node ${active === node.id ? 'is-active' : ''}`}
           onClick={() => { setOpen(!open); onSelect(node) }}>
        {node.children?.length
          ? <span className={`xx-tree-caret ${open ? 'is-open' : ''}`}>▶</span>
          : <span style={{width:14}} />}
        <span>{node.name}</span>
      </div>
      {open && node.children &&
        <div className="xx-tree-children">
          {node.children.map(c => <TreeNode key={c.id} node={c} ... />)}
        </div>}
    </>
  )
}
