구현 패턴

// 백엔드 - JOIN 으로 라벨 함께 반환
"LEFT JOIN co_common_code cc "
" ON cc.common_cd_group='COMMON' "
" AND cc.common_cd_id='work_status_cd' "
" AND cc.common_cd=t.work_status_cd "
// → 응답 필드: work_status_label

// 프론트 - JOIN 라벨 우선 + fallback 클라이언트 매핑
const STATUS_OPTIONS = [
  { cd: '1', label: '작업요청' }, { cd: '2', label: '작업접수' },
  { cd: '3', label: '작업시작' }, { cd: '4', label: '작업완료' },
  { cd: '5', label: '작업지연' }, { cd: '6', label: '작업이관' },
]
const statusLabel = (cd) => STATUS_OPTIONS.find(o => o.cd === cd)?.label || cd || '-'

// 그리드 셀
{ key: 'work_status_cd', header: '상태', width: 90, align: 'center',
  render: (r) => (
    <span className={`xx-pill ${statusToClass(r.work_status_cd)}`}>
      {r.work_status_label || statusLabel(r.work_status_cd)}
    </span>
  ),
}
