  const [over, setOver] = useState(false)
  return (
    <div
      className={`xx-dropzone ${over ? 'is-over' : ''}`}
      onDragOver={(e) => { e.preventDefault(); setOver(true) }}
      onDragLeave={() => setOver(false)}
      onDrop={(e) => { e.preventDefault(); setOver(false); onFiles(Array.from(e.dataTransfer.files)) }}

가이드

  • 파일별 아이콘은 확장자 매핑 (pdf, xlsx, jpg/png, etc.)
  • 파일명 너무 길면 ellipsis. 호버 시 툴팁으로 전체 노출
  • 이미지 (.jpg/.png) 는 클릭 시 라이트박스로 큰 미리보기 (점검결과 사진 등)
  • 현장 모바일 업로드는 카메라 캡처 input: <input type="file" accept="image/*" capture="environment" />
