import Grid from '../M000000000/shared/Grid'

const cols = [
  { key: 'work_order_id', header: '오더번호', width: 130 },
  { key: 'work_order_name', header: '제목' },
  { key: 'user_name', header: '요청자', width: 110 },
  { key: 'work_status_cd', header: '상태', width: 90, align: 'center',
    render: (r) => <span className={`xx-pill ${stCls(r.work_status_cd)}`}>{r.work_status_label}</span>,
  },
  { key: 'work_req_date', header: '요청일', width: 110, render: (r) => fmtDate(r.work_req_date) },
]

<Grid<Row>
  columns={cols} rows={rows}
  rowKey={(r) => r.work_order_id}
  onRowClick={(r) => setActive(r)}
  selectable
  pageSize={pageSize} onPageSizeChange={setPageSize}
  emptyMessage="검색 결과가 없습니다."
/>

컬럼 정렬·너비 규칙

  • 코드/숫자/날짜는 width 명시 (예: 90, 110, 130). 가변은 제목/내용 1 개만.
  • 상태 칩 셀은 align: 'center'
  • 금액/수량은 align: 'right' + font-variant-numeric: tabular-nums
  • 긴 텍스트는 자동 ellipsis + 툴팁 (셀 hover)
