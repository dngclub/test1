서명 완료 (이미지 미리보기)

// canvas 캡처 → base64 data URL → 서버 전송
function SignaturePad({ value, onChange }) {
  const ref = useRef(null)
  useEffect(() => {
    const c = ref.current; const ctx = c.getContext('2d')
    ctx.lineWidth = 2; ctx.lineCap = 'round'; ctx.strokeStyle = '#111'
    let drawing = false; let last = { x:0, y:0 }
    const pos = (e) => {
      const r = c.getBoundingClientRect()
      const t = e.touches?.[0] || e
      return { x: t.clientX - r.left, y: t.clientY - r.top }
    }
    const start = (e) => { drawing = true; last = pos(e) }
    const move = (e) => {
      if (!drawing) return
      e.preventDefault()
      const p = pos(e)
      ctx.beginPath(); ctx.moveTo(last.x, last.y); ctx.lineTo(p.x, p.y); ctx.stroke()
      last = p
    }
    const end = () => { drawing = false; onChange(c.toDataURL('image/png')) }
    c.addEventListener('mousedown', start); c.addEventListener('mousemove', move); c.addEventListener('mouseup', end)
    c.addEventListener('touchstart', start); c.addEventListener('touchmove', move); c.addEventListener('touchend', end)
    return () => { /* cleanup */ }
  }, [])
  return <canvas ref={ref} width={400} height={140} className="xx-sign-canvas" />
}

가이드

  • 고정 크기 400×140 canvas. PNG base64 로 저장.
  • 모바일 점검·작업허가에서 가장 자주 사용됨 → 터치 이벤트 필수.
  • 저장 후 view 모드에서는 <img src={value} /> 로 표시.
  • 서명자 정보(이름, 시각, 디바이스) 함께 기록 권장.
