const [open, setOpen] = useState(false)
const ref = useRef(null)

// 외부 클릭 시 닫기
useEffect(() => {
  const fn = (e) => { if (!ref.current?.contains(e.target)) setOpen(false) }
  if (open) document.addEventListener('click', fn)
  return () => document.removeEventListener('click', fn)
}, [open])

<div className="xx-menu-wrap" ref={ref}>
  <button className="eg-btn eg-btn-default" onClick={()=>setOpen(!open)}>⋮ 더보기</button>
  {open &&
    <div className="xx-menu">
      <div className="xx-menu-item" onClick={()=>{ edit(); setOpen(false) }}>✏️ 수정</div>
      <div className="xx-menu-item is-danger" onClick={...}>🗑 삭제</div>
    </div>}
</div>

가이드

  • 한두개 액션은 직접 버튼으로 노출. 3개 이상이거나 화면이 좁을 때 드롭다운.
  • 위험 액션은 is-danger + xx-menu-sep 으로 분리
  • 외부 클릭으로 닫히게 useEffect 처리 필수
